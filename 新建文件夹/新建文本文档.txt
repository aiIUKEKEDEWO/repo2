	<head>

		<title></title>
		<script type="text/javascript" src="./js/vue.js"></script>
	</head>
	<style>
		.aClass{
			color: aqua;
		}
		.bClass{
			color: black;
		}
	</style>
	<body>
		<div id="demo">
			<h2>1.class绑定：：class='XXX'</h2>
			<p :class="a">1</p>
			<p :class="{aClass:isA,bClass:isB}">123</p>
			<h2>2.style绑定</h2>
			<p :style="{color:activeColor,fontSize:fontSize+'px'}">111112121212</p>
			<button type="button" @click="update">更新</button>
		</div>
		<script type="text/javascript">
			new Vue({
				el:'#demo',
				data:{
					a:'aClass',
					isA:true,
					isB:false,
					activeColor:'red',
					fontSize:'20'
				},
				methods:{
					update(){
						this.a='bClass'
						this.isA=false,
						this.isB=true,
						this.activeColor='black',
						this.fontSize=40
						
					}
				}
			})
		</script>
	</body>
